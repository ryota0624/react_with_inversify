"use strict";
var todoAction_1 = require("../action/todoAction");
var Todo_1 = require("../model/Todo");
var inversifyConfig_1 = require("../inversifyConfig");
function addTodoRequest(text) {
    var todo = new Todo_1.Todo(text, new Date, Math.floor(Math.random() * 100));
    setTimeout(function () { return inversifyConfig_1.kernel.get(inversifyConfig_1.TYPES.Dispatcher).dispatch(todoAction_1.addTodo(todo)); }, 100);
}
exports.addTodoRequest = addTodoRequest;
