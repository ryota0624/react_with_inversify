TodoDto.js

"use strict";
var TodoDto = (function () {
    function TodoDto(name, id) {
        this.name = name;
        this.id = id;
    }
    return TodoDto;
}());
exports.TodoDto = TodoDto;
function todo2Dto(todo) {
    return new TodoDto(todo.name.value, todo.id);
}
exports.todo2Dto = todo2Dto;
